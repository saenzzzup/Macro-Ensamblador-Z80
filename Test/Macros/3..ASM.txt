CARGA: MACRO #a1, #a2, #a3
	LD #a1, A
	LD #a2, B
	LD #a3, C
	MEND

CAMBIA: MACRO #a4, #a5
	PUSH #a4
	PUSH #a5
	CARGA2: MACRO
		LD #a4, A
	MEND
	POP #a4
	POP #a5
	MEND

INICIO: LD A, (20H)
	ADD A, B
	INC C
	CARGA E, H, L
	DEC B
	CARGA2
	SLA B
	CAMBIA HL, DE
	CARGA (200H), (IX), (HL)
	HALT
	END INICIO
